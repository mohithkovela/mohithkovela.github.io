import java.util.Scanner;
import org.jsoup.Jsoup;
import org.jsoup.nodes.Document;
import org.jsoup.nodes.Element;
import org.jsoup.select.Elements;

class Dictionary
{
	public static void main(String args[])
	{
		try
		{
			Scanner scanner = new Scanner(System.in);
			System.out.println("Which English word meaning do you want? ");
			String word = scanner.next();
			Document document;
			document = Jsoup.connect("http://www.shabdkosh.com/te/translate?e=" + word + "&l=te").get();
			int counter = 0;	
			Elements wordtag = document.select("a[class]");
			for (Element element : wordtag)
			{
				counter++;
				if(counter == 20)
				{
					System.out.println("Telugu meaning: "element.text());
				}
			}
		}
		catch (Exception e)
		{
			e.printStackTrace();
		}
	}
}

Program to send an e-mail.

import java.util.Date;
import java.util.Properties;
import javax.mail.Message;
import javax.mail.PasswordAuthentication;
import javax.mail.Session;
import javax.mail.Transport;
import javax.mail.internet.InternetAddress;
import javax.mail.internet.MimeMessage;
import javax.mail.*;
import java.util.Scanner;
import java.io.Console;
public class SendEmail
{
   public static void main(String []args)
   {
      Scanner input = new Scanner(System.in);
      try
      {
         Console cnsl = System.console();
         Properties propsObject = new Properties();
         propsObject.put("mail.smtp.host", "smtp.gmail.com");
         propsObject.put("mail.smtp.auth", "true");
         propsObject.put("mail.debug", "true"); 
         propsObject.put("mail.smtp.starttls.enable", "true");
         propsObject.put("mail.smtp.port", "465");
         propsObject.put("mail.smtp.socketFactory.port", "465");
         propsObject.put("mail.smtp.socketFactory.class", "javax.net.ssl.SSLSocketFactory");
         propsObject.put("mail.smtp.socketFactory.fallback", "false");
         System.out.print("Enter your gmail id: ");
         final String sourceAddress = input.next();
         System.out.print("Enter your password: ");
         char pwd[] = cnsl.readPassword();
         final String password = new String(pwd);
         input.nextLine(); 
         System.out.print("Enter the subject: ");
         String subject = input.nextLine();
         System.out.println("Write body of the mail: ");
         String text = input.nextLine(); 
         System.out.print("Enter recipient's e-mail id: ");
         String destinationAddress = input.next();
         Session mailSession = Session.getInstance(propsObject, new Authenticator()
         {
            public PasswordAuthentication getPasswordAuthentication() 
            {
               return new PasswordAuthentication(sourceAddress, password);
            }
         });
         mailSession.setDebug(true);
         Message msgObject = new MimeMessage(mailSession);
         msgObject.setFrom(new InternetAddress(sourceAddress));
         msgObject.setRecipients(Message.RecipientType.TO,InternetAddress.parse(destinationAddress));
         msgObject.setSentDate(new Date());
         msgObject.setSubject(subject);
         msgObject.setText(text);
         Transport.send(msgObject);
      }
      catch(Exception E)
      {
        System.out.println( "Oops! Something went wrong.");
        System.out.println(E);
      }
   }
}

Program to convert color image into grayscale.

import java.awt.*;
import java.awt.image.BufferedImage;
import java.util.*;
import java.text.*;
import java.io.*;
import javax.imageio.ImageIO;

public class GrayScale 
{

   private BufferedImage image;
   private int width;
   private int height;
   
   public GrayScale(String sourceImage) {
   
      try 
      {
         image = ImageIO.read(new File (sourceImage));
         width = image.getWidth();
         height = image.getHeight();
      
         for(int i=0; i < height; i++)
         {
      
            for(int j=0; j < width; j++)
            {
            
               Color c = new Color(image.getRGB(j, i));
               int red = (int)(c.getRed() * 0.299);
               int green = (int)(c.getGreen() * 0.587);
               int blue = (int)(c.getBlue() *0.114);
               Color newColor = new Color(red+green+blue, red+green+blue, red+green+blue);
               image.setRGB(j,i,newColor.getRGB());
            }
         }

         Date now = new Date();
         SimpleDateFormat ft = new SimpleDateFormat ("yyyyMMddhhmmss");
         String fileName =  ft.format(now) + ".jpg";
         File ouptut = new File(fileName);
         ImageIO.write(image, "jpg", ouptut);
         System.out.println("Image is converted into grayscale and saved as " + fileName + ".");
      } 
      catch (Exception e) 
      {
         e.printStackTrace();
      }
   }
   static public void main(String args[]) throws Exception 
   {
      GrayScale obj = new GrayScale(args[0]);
   }
}

Program to access data from XML file.

import javax.xml.parsers.DocumentBuilderFactory;
import javax.xml.parsers.DocumentBuilder;
import org.w3c.dom.Document;
import org.w3c.dom.NodeList;
import org.w3c.dom.Node;
import org.w3c.dom.Element;
import java.io.File;
import java.util.Scanner;
class ReadXmlFile
{
	public static void main(String argv[])
	{
		Scanner input = new Scanner(System.in);
   
		try
		{

			File objXmlFile = new File("JavaXML.xml");
			DocumentBuilderFactory objFactory = DocumentBuilderFactory.newInstance();
			DocumentBuilder objBuilder = objFactory.newDocumentBuilder();
			Document docObj = objBuilder.parse(objXmlFile);
			docObj.getDocumentElement().normalize();
			System.out.print("Enter Product ID: ");
			String productID = input.next();
			NodeList nList = docObj.getElementsByTagName("Products");
			for(int temp = 0; temp < nList.getLength(); temp++)
			{
				Node nNode = nList.item(temp);
				if (nNode.getNodeType() == Node.ELEMENT_NODE)
				{
					Element eElement = (Element) nNode;
					if(eElement.hasAttribute("ProductId") && eElement.getAttribute("ProductId").equals(productID))
					{
						System.out.println("Product Name : " + eElement.getElementsByTagName("ProductName").item(0).getTextContent());
						System.out.println("Product Price : " + eElement.getElementsByTagName("ProductPrice").item(0).getTextContent());
						System.out.println("Product Description : " + eElement.getElementsByTagName("ProductDescription").item(0).getTextContent());
					}
				}
			}
		}
		catch(Exception e)
		{
			System.out.println("Product could not be found.");
		}
 	}
}

Program to access data from JSON file.

import java.util.Iterator;
import org.json.simple.JSONArray;
import org.json.simple.JSONObject;
import org.json.simple.parser.JSONParser;
import java.util.Scanner;
import java.io.FileReader; 	

class JSONReader
{
	public static void main(String []args)
	{
		try
		{
			JSONParser parser = new JSONParser();
			Object object = parser.parse(new FileReader("Products.JSON"));
			JSONObject jsonObject = (JSONObject) object;
			Scanner input = new Scanner(System.in);
			System.out.print("Enter product id: ");
			String productId = input.next();
			String productAttributes[] = {"Product name","Product price","Stock quantity","Product description"};
			JSONArray productAttributeValues = (JSONArray) jsonObject.get(productId);
			Iterator iterator = productAttributeValues.iterator();
			int counter = 0;
			while (iterator.hasNext())
			{	
				System.out.println(productAttributes[counter++]+": "+iterator.next());
			}
		}
		catch(Exception e)
		{
			System.out.println("Product not found.");
		}
	}
}
