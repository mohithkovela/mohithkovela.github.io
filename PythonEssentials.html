Program to send an e-mail.

	      import smtplib
	      from emailCredentials import emailaddress, password
	      tomail = input("enter email address: ")
	      subject = input("Enter subject: ")
	      text = input("Enter your message here: ")
	      message = 'Subject: {}\n\n{}'.format(subject, text)
	      try:
			mail = smtplib.SMTP('smtp.gmail.com', 587)
			mail.ehlo()
			mail.starttls()
			mail.login(emailaddress, password)
			mail.sendmail(emailaddress, tomail, message)
			mail.close()
	      except:
			print("Something went wrong. Unable to send e-mail.")

Program to convert color image to grayscale.

	      	import sys
	      	from datetime import *
	      	from PIL import Image
	      	from datetime import datetime
	      	filePath = sys.argv[1]
	      	imageObj = Image.open(filePath)
	      	time = datetime.now()
	      	imgName = str(date.today()) + str(time.hour) + str(time.minute) + str(time.second) + '.jpg'
	      	imageObj.convert(mode = 'L').save(imgName)

Program to call ps command and store results in database.

	      import os
	      import sqlite3 as lite
	      import unicodecsv
	      databaseName = "SystemProcessStatus.db"
	      databaseObject = lite.connect(databaseName)
	      cursorObject = databaseObject.cursor()
	      cmd = 'tasklist /v'
	      cmdObject = os.popen(cmd).read()
	      cursorObject.execute("CREATE TABLE Process_Status(Image_Name Text, PID Text, Session_Name Text, Session Text, Mem_Usage Text, Status Text, User_Name Text, CPU_Time Text, Window_Title Text)")
	      with open('ProcessStatus.txt', "w") as file:
	      	data=[row.split('\t') for row in file.readlines()]
	      	cursorObject.executemany("INSERT INTO Process_Status(Image_Name, PID, Session_Name, Session, Mem_Usage, Status, User_Name, CPU_Time, Window_Title) VALUES (?, ?, ?);", data)
	      databaseObject.commit()
	      print("System process status is stored in a database.")
